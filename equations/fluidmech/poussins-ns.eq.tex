\begin{align}
  \left(\partial\vec{v}/\partial{t}\right)^n &%
      = \vec{f}(\vec{v}^n, \rho^n)
      = \vec{\tilde{f}}(\vec{v}^n, \rho^n) - \nabla p^n / \rho^n
    \label{eq:poussins_ns_momentum}
    \\[1ex]
    %%%
    \left(\partial\rho/\partial{t}\right)^n
      &= g(\vec{v}^n, \rho^n)
    \label{eq:poussins_ns_mass}
\end{align}
